121 // Generational replacement - Add best elites and children
122 template <class POPULATIONTYPE>
123 bool FloatReplacement<POPULATIONTYPE>::generational(POPULATIONTYPE &population, POPULATIONTYPE &children)
124 {
125  // Used to store the next population
126  POPULATIONTYPE newPopulation;
127 
128  // Sort populations depending on problem type
129  if (isMinimizationProblem)
130  {
131  std::sort(std::begin(population.individuals), std::end(population.individuals), sortIndividualMin);
132  std::sort(std::begin(children.individuals), std::end(children.individuals), sortIndividualMin);
133  }
134  else
135  {
136  std::sort(std::begin(population.individuals), std::end(population.individuals), sortIndividualMax);
137  std::sort(std::begin(children.individuals), std::end(children.individuals), sortIndividualMax);
138  }
139 
140  // Make sure our new population is the correct size
141  newPopulation.individuals.resize(population.individuals.size());
142 
143  // Get number of individuals to keep with elitism
144  int elitismSize = floor(population.individuals.size() * elitismRate);
145 
146  // Add the elites
147  for (int i = 0; i < elitismSize; ++i)
148  {
149  newPopulation.individuals.at(i) = population.individuals.at(i);
150  }
151 
152  // Add the children
153  for (int i = elitismSize; i < population.individuals.size(); ++i)
154  {
155  newPopulation.individuals.at(i) = children.individuals.at(i - elitismSize);
156  }
157 
158  // Set the population to the new population
159  population = newPopulation;
160 
161  return true;
162 }
