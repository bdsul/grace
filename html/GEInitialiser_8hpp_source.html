274  {

276 
277  // Add an extra individual to make the population even
278  ++this->populationSize;

282  unsigned int minimumDepth = grammarFile.getStartRule()->getMinimumDepth();

284  // Get range of depths - Min and Max inbetween
285  if (minimumDepth > sensibleMaxDepth)

288  std::cout << "Error: Minimum rhh depth: " << minimumDepth << std::endl;

290  std::cout << "Error: Exiting..." << std::endl;

292  exit(EXIT_FAILURE);

297 
298  for (int depth = minimumDepth; depth <= sensibleMaxDepth; depth++)

314 
315  // Loop over the population
316  for (int q = 0; q < quotient; ++q)
317  {
318  for (unsigned int depth : depths)

417 template <class POPULATIONTYPE>
418 bool GEInitialiser<POPULATIONTYPE>::createSubTree(GenomeType &individual, int maxDepth, int type, Symbol &currentSymbol, int currentDepth)
419 {
420  // Is the current depth greater than the maximum depth?
421  if (currentDepth > maxDepth)
422  {
423  return false;
424  }
425 
426  // If the symbol is a terminal, return
427  if (currentSymbol.getType() == Symbol::TerminalSymbol)
428  {
429  return true;
430  }
431 
432  // Symbol is a non-terminal. Get the corrosponding rule
433  const CFRule *currentRule = individual.grammar.findRule(currentSymbol);
434 
435  // Was the rule found?
436  if (currentRule == nullptr)
437  {
438  std::cout << "Error: Unable to find Rule for Symbol " << currentSymbol.getValue() << std::endl;
439  exit(EXIT_FAILURE);
440  }
441 
442  // Get the possible choices
443  std::vector<unsigned int> validChoicesIndex;
444 
445  for (int index = 0; index < currentRule->rhs.size(); ++index)
446  {
447  // Don't exceed the maximum depth
448  if (currentDepth + currentRule->rhs.at(index).getMinimumDepth() > maxDepth)
449  {
450  continue;
451  }
452 
453  // Check if we're at the max depth
454  if (currentDepth == maxDepth)
455  {
456  // We can only return choices that exclusively contain terminals
457  bool isValid = true;
458 
459  // Does the choice contain a non-terminal?
460  for (std::shared_ptr<Symbol> symbol : currentRule->rhs.at(index).symbols)
461  {
462  if (symbol->getType() == Symbol::NonTerminalSymbol)
463  {
464  isValid = false;
465  break;
466  }
467  }
468 
469  // Is the choice valid?
470  if (isValid == false)
471  {
472  continue;
473  }
474 
475  // Add the choice to our possible choices
476  validChoicesIndex.push_back(index);
477  }
478 
479  // For FULL, we have to have at least one non-terminal
480  // We also need to check that the choice can map in the desired depth
481  // Although I don't know if the above check removes the need for this
482  if (type == FULL)
483  {
484  // We can only return choices that contain nonterminals
485  // They also need to be mappable in at the desired depth
486  bool isValid = false;
487  for (std::shared_ptr<Symbol> symbol : currentRule->rhs.at(index).symbols)
488  {
489  if (symbol->getType() == Symbol::NonTerminalSymbol)
490  {
491  isValid = true;
492  break;
493  }
494  }
495 
496  // If the choice contains a non-terminal, add it to our possible choices
497  if (isValid == true)
498  {
499  validChoicesIndex.push_back(index);
500  }
501  }
502  else
503  {
504  // Grow method - We can choose either
505  validChoicesIndex.push_back(index);
506  }
507  }
508 
509  // Is there a choice to choose?
510  if (validChoicesIndex.empty())
511  {
512  std::cout << "Error: No Choices possible" << std::endl;
513  return false;
514  }
515 
516  unsigned int chosenChoiceIndex = 0;
517 
518  if (currentRule->rhs.size() == 1)
519  {
520  // If there's only one choice, we don't need a codon
521  }
522  else
523  {
524  unsigned int codon;
525  std::uniform_int_distribution<> codonDistribution(0, (UINT8_MAX / currentRule->rhs.size()));
526 
527  // If there's multiple choices, we need a codon
528  if (validChoicesIndex.size() > 1)
529  {
530  // We need to choose a possible choice
531  std::uniform_int_distribution<> choiceDistribution(0, validChoicesIndex.size() - 1);
532  chosenChoiceIndex = validChoicesIndex.at(choiceDistribution(this->rng));
533  }
534  else
535  {
536  // Add the only valid choice.
537  // A codon will still be required as there is more than one choice,
538  // even if some choices are invalid
539  chosenChoiceIndex = validChoicesIndex.at(0);
540  }
541 
542  // Add the codon to the genome
543  codon = (codonDistribution(this->rng) * currentRule->rhs.size()) + chosenChoiceIndex;
544  individual.genotype.push_back(codon);
545  }
546 
547  // For each symbol in the chosen production, recursively call this function
548  for (std::shared_ptr<Symbol> symbol : currentRule->rhs.at(chosenChoiceIndex).symbols)
549  {
550  bool isValid = createSubTree(individual, maxDepth, type, *symbol, currentDepth + 1);
551  if (isValid == false)
552  {
553  return false;
554  }
555  }
556 
557  return true;
558 }
